	}
	
	potential = {
		has_modifier = "apsr7_planet_modifier"

		}
	}

		potential = {
			exists = owner

			}
		}
		modifier = {

		}
	}	
		
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				is_gestalt = no
			}
		}
		modifier = {
			job_apsr7_soldier_normal_add = 3
		}
	}
	
	resources = {
		category = planet_apsr_buildings
