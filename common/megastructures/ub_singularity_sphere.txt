singularity_sphere_0 = {
	entity = "dyson_sphere_phase_01_entity"
	portrait = "GFX_megastructure_construction_background"
	build_time = 3600
	resources = {
		category = megastructures
		cost = {
			energy = 500000
			minerals = 500000
			food = 500000
			consumer_goods = 200000
			alloys = 100000
			exotic_gases = 50000
			rare_crystals = 50000
			volatile_motes = 50000
			sr_dark_matter = 20000
			sr_living_metal = 20000
			sr_zro = 20000
			nanites = 10000
		}
		upkeep = {
			energy = 100
			minerals = 100
			food = 100
			consumer_goods = 50
			alloys = 25
			exotic_gases = 10
			rare_crystals = 10
			volatile_motes = 10
			sr_dark_matter = 5
			sr_living_metal = 5
			sr_zro = 5
			nanites = 2
		}
	}

	country_modifier = {
		planet_jobs_produces_mult = 0.25
		planet_buildings_produces_mult = 0.25
		starbase_constructions_produces_mult = 0.25
		station_gatherers_produces_mult = 0.25
		station_researchers_produces_mult = 0.25
	}

	prerequisites = { "tech_mega_engineering" }
	upgrade_from = { singularity_sphere_ruined }
	construction_blocks_and_blocked_by = none

	possible = {
		hidden_trigger = {
			exists = starbase
		}
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
		custom_tooltip = {
			fail_text = "requires_surveyed_system"
			NOT = {
				any_system_planet = {
					is_surveyed = {
						who = prev.from
						status = no
					}
				}
			}
		}
	}

	potential = {
		is_ai = no
	}

	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "must_build_around_star"
				is_star = yes
			}
			custom_tooltip = {
				fail_text = "requires_no_anomaly"
				NOT = { has_anomaly = yes }
			}
		}
	}

	ai_weight = {
		factor = 0
	}

	on_build_start = {
		set_star_flag = singularity_sphere_construction
	}
	on_build_cancel = {
		remove_star_flag = singularity_sphere_construction
	}
	on_build_complete = {
		remove_star_flag = singularity_sphere_construction
		set_star_flag = singularity_sphere_built
		from = { country_event = { id = ub_singularity_sphere.0 } }
		fromfrom = {
			solar_system.star = {
				create_ambient_object = {
					type = "crisis_sphere_1"
					location = this
				}
				last_created_ambient_object = {
					set_ambient_object_flag = crisis_sphere_1_system_effect
					set_location = {
						target = prev
						distance = 0
						angle = random
					}
				}
			}
			set_megastructure_flag = singularity_sphere
		}
	}
}
