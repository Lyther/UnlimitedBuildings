# Trade buildings

ub_building_galactic_stock_exchange = {
	icon = building_galactic_stock_exchange
	base_buildtime = @b2_time

	category = trade

	potential = {
		NOR = { 
			has_modifier = resort_colony 
			has_modifier = slave_colony 
			has_modifier = penal_colony	
		}
		has_building = building_galactic_stock_exchange
		owner = { is_ai = no }
	}

	allow = {
		has_major_upgraded_capital = yes
		if = {
			limit = {
				exists = orbital_defence
			}
			orbital_defence = {
				exists = starbase
				starbase = {
					NOR = {
						has_starbase_building = ring_galactic_stock_exchange
						is_starbase_building_building = ring_galactic_stock_exchange
					}
				}
			}
		}
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
			has_modifier = resort_colony
			has_modifier = penal_colony	
		}
	}


	planet_modifier = {
		job_merchant_add = 1
		trade_value_mult = 0.20
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b2_minerals
			rare_crystals = @b2_rare_cost
		}
		upkeep = {
			energy = @b2_upkeep
			rare_crystals = @b2_rare_upkeep
		}
	}

	prerequisites = {
		"tech_galactic_markets"
	}

	triggered_desc = {
		text = job_merchant_effect_desc
	}

	ai_resource_production = {
		energy = 5
		trigger = {
			solar_system = {
				has_system_trade_value >= 50
			}
		}
	}

	ai_resource_production = {
		energy = 3
		trigger = {
			solar_system = {
				has_system_trade_value >= 80
			}
		}
	}
}
